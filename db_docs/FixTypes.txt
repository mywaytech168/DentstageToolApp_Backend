CREATE TABLE fix_types (
    FixType VARCHAR(50) NOT NULL PRIMARY KEY,
    FixTypeName VARCHAR(100) NOT NULL
) ENGINE=InnoDB;

    ('other', '其他')
ON DUPLICATE KEY UPDATE FixTypeName = VALUES(FixTypeName);
