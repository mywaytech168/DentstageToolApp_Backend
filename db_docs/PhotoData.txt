CREATE TABLE PhotoData (
    PhotoUID            VARCHAR(100) PRIMARY KEY,
    QuotationUID        VARCHAR(100),
    RelatedUID          VARCHAR(100),
    Posion              VARCHAR(50),
    Comment             TEXT,
    Photo_shape         VARCHAR(50),
    Photo_shape_other   VARCHAR(50),
    Photo_shape_show    VARCHAR(50),
    Cost                DECIMAL(10,2),
    Flag_Finish         BOOLEAN,
    Finish_Cost         DECIMAL(10,2),
    -- 維修進度百分比，新流程依此計算實收金額。
    MaintenanceProgress DECIMAL(5,2),
    -- 照片階段（例如 before / after），供維修後照片綁定使用。
    PhotoStage          VARCHAR(20),
    -- Fix_Type 改儲存固定鍵值（dent、beauty、paint、other），預設長度調整為 100 以容納描述文字。
    Fix_Type            VARCHAR(100)
);
